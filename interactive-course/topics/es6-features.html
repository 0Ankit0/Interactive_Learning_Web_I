Let & Const

Variable Declaration Comparison

// ES5 - var (function scope)
var name = 'John';
if (true) {
    var name = 'Jane'; // Overwrites outer variable
}
console.log(name); // 'Jane'

// ES6 - let (block scope)
let age = 25;
if (true) {
    let age = 30; // Different variable
    console.log(age); // 30
}
console.log(age); // 25

// ES6 - const (block scope, immutable binding)
const PI = 3.14159;
// PI = 3.14; // Error: Assignment to constant variable

var

Function scope, can be redeclared, hoisted

var x = 1;
var x = 2; // OK

let

Block scope, cannot be redeclared in same scope

let y = 1;
// let y = 2; // Error

const

Block scope, cannot be reassigned

const z = 1;
// z = 2; // Error

Arrow Functions

Arrow Function Syntax

// Traditional function
function add(a, b) {
    return a + b;
}

// Arrow function
const add = (a, b) => a + b;

// Arrow function with body
const multiply = (a, b) => {
    const result = a * b;
    return result;
};

// Single parameter (parentheses optional)
const square = x => x * x;

// No parameters
const getRandom = () => Math.random();

// Object return (wrap in parentheses)
const createUser = (name, age) => ({ name, age });

// In callbacks
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);

               Template Literals
            


            

              
Template Literal Examples

              

                
// ES5 string concatenation
var name = 'John';
var greeting = 'Hello, ' + name + '! Today is ' + new Date().toDateString();

// ES6 template literals
const name = 'John';
const greeting = `Hello, ${name}! Today is ${new Date().toDateString()}`;

// Multi-line strings
const html = `
<div class="user">
    <h2>${name}</h2>
    <p>Welcome back!</p>
</div>
`;

// Expression interpolation
const a = 5, b = 10;
console.log(`The sum of ${a} and ${b} is ${a + b}`);

// Tagged templates
function highlight(strings, ...values) {
    return strings.reduce((result, string, i) =>
        result + string + (values[i] ? `<strong>${values[i]}</strong>` : ''), '');
}

const result = highlight`Hello ${name}, you have ${messages.length} messages`;

              

            

          


          

            

               Destructuring
            


            

              
Array Destructuring

              

                
// Array destructuring
const numbers = [1, 2, 3, 4, 5];

// Basic destructuring
const [first, second, third] = numbers;
console.log(first, second, third); // 1, 2, 3

// Skip elements
const [a, , c] = numbers;
console.log(a, c); // 1, 3

// Rest operator
const [head, ...tail] = numbers;
console.log(head, tail); // 1, [2, 3, 4, 5]

// Default values
const [x = 0, y = 0] = [];
console.log(x, y); // 0, 0

// Swapping values
let a = 1, b = 2;
[a, b] = [b, a];
console.log(a, b); // 2, 1

              

            


            

              
Object Destructuring

              

                
// Object destructuring
const user = {
    name: 'John',
    age: 30,
    email: 'john@example.com',
    address: {
        city: 'New York',
        country: 'USA'
    }
};

// Basic destructuring
const { name, age } = user;
console.log(name, age); // 'John', 30

// Alias variables
const { name: userName, age: userAge } = user;
console.log(userName, userAge); // 'John', 30

// Default values
const { name, role = 'user' } = user;
console.log(name, role); // 'John', 'user'

// Nested destructuring
const { address: { city, country } } = user;
console.log(city, country); // 'New York', 'USA'

// Rest operator
const { name, ...userInfo } = user;
console.log(userInfo); // { age: 30, email: '...', address: {...} }

// Function parameters
function greet({ name, age }) {
    return `Hello ${name}, you are ${age} years old`;
}

console.log(greet(user)); // 'Hello John, you are 30 years old'

              

            

          


          

            

               Spread & Rest Operators
            


            

              
Spread Operator Examples

              

                
// Array spreading
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2];
console.log(combined); // [1, 2, 3, 4, 5, 6]

// Copying arrays
const original = [1, 2, 3];
const copy = [...original];

// Adding elements
const withExtras = [...original, 4, 5];

// Object spreading
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const merged = { ...obj1, ...obj2 };
console.log(merged); // { a: 1, b: 2, c: 3, d: 4 }

// Function calls
function sum(a, b, c) {
    return a + b + c;
}
const numbers = [1, 2, 3];
console.log(sum(...numbers)); // 6

// String to array
const str = 'hello';
const chars = [...str];
console.log(chars); // ['h', 'e', 'l', 'l', 'o']

              

            


            

              
Rest Operator Examples

              

                
// Rest in function parameters
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4)); // 10

// Rest with destructuring
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first, second, rest); // 1, 2, [3, 4, 5]

const { a, b, ...others } = { a: 1, b: 2, c: 3, d: 4 };
console.log(a, b, others); // 1, 2, { c: 3, d: 4 }

// Rest in arrow functions
const multiply = (multiplier, ...numbers) =>
    numbers.map(num => num * multiplier);

console.log(multiply(2, 1, 2, 3)); // [2, 4, 6]

              

            

          


          

            

               Promises & Async/Await
            


            

              
Promise Creation and Usage

              

                
// Creating a Promise
const fetchData = () => {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const success = Math.random() > 0.5;
            if (success) {
                resolve('Data fetched successfully');
            } else {
                reject('Failed to fetch data');
            }
        }, 1000);
    });
};

// Using Promises
fetchData()
    .then(result => {
        console.log('Success:', result);
        return result.toUpperCase();
    })
    .then(processed => {
        console.log('Processed:', processed);
    })
    .catch(error => {
        console.error('Error:', error);
    })
    .finally(() => {
        console.log('Operation completed');
    });

// Promise.all - run multiple promises in parallel
const promise1 = Promise.resolve('First');
const promise2 = new Promise(resolve => setTimeout(() => resolve('Second'), 1000));
const promise3 = fetch('https://api.example.com/data').then(res => res.json());

Promise.all([promise1, promise2, promise3])
    .then(results => console.log(results))
    .catch(error => console.error(error));

              

            


            

              
Async/Await Syntax

              

                
// Async function declaration
async function fetchUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        if (!response.ok) {
            throw new Error('Failed to fetch user');
        }
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Error fetching user:', error);
        throw error; // Re-throw to allow caller to handle
    }
}

// Async arrow function
const fetchUserData = async (userId) => {
    try {
        const response = await fetch(`/api/users/${userId}`);
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Error:', error);
        return null; // Return default value on error
    }
};

// Using async functions
async function displayUser(userId) {
    try {
        const user = await fetchUserData(userId);
        if (user) {
            console.log('User:', user.name);
        }
    } catch (error) {
        console.error('Failed to display user:', error);
    }
}

// Top-level await (in modules)
const user = await fetchUserData(123);
console.log(user);

              

            

          


          

            

               Modules
            


            

              
Exporting from Modules

              

                
// math.js
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// Default export
export default function square(x) {
    return x * x;
}

// Alternative default export syntax
export { square as default };

// Named exports
const subtract = (a, b) => a - b;
const divide = (a, b) => a / b;

export { subtract, divide };

              

            


            

              
Importing from Modules

              

                
// Importing named exports
import { add, multiply, PI } from './math.js';

// Importing default export
import square from './math.js';

// Importing with aliases
import { add as sum, multiply as product } from './math.js';

// Importing everything
import * as MathUtils from './math.js';
console.log(MathUtils.add(2, 3)); // 5

// Mixed imports
import square, { add, PI } from './math.js';

// Dynamic imports
const module = await import('./math.js');
console.log(module.add(1, 2));
