Let & Const

Variable Declaration Comparison

// ES5 - var (function scope)
var name = 'John';
if (true) {
    var name = 'Jane'; // Overwrites outer variable
}
console.log(name); // 'Jane'

// ES6 - let (block scope)
let age = 25;
if (true) {
    let age = 30; // Different variable
    console.log(age); // 30
}
console.log(age); // 25

// ES6 - const (block scope, immutable binding)
const PI = 3.14159;
// PI = 3.14; // Error: Assignment to constant variable

var

Function scope, can be redeclared, hoisted

var x = 1;
var x = 2; // OK

let

Block scope, cannot be redeclared in same scope

let y = 1;
// let y = 2; // Error

const

Block scope, cannot be reassigned

const z = 1;
// z = 2; // Error

const createUser = (name, age) => ({ name, age });

// In callbacks
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);

               Destructuring
            


            

              
Array Destructuring

              

                
// Array destructuring
const numbers = [1, 2, 3, 4, 5];

// Basic destructuring
const [first, second, third] = numbers;
console.log(first, second, third); // 1, 2, 3

// Skip elements
const [a, , c] = numbers;
console.log(a, c); // 1, 3

// Rest operator
const [head, ...tail] = numbers;
console.log(head, tail); // 1, [2, 3, 4, 5]

// Default values
const [x = 0, y = 0] = [];
console.log(x, y); // 0, 0

// Swapping values
let a = 1, b = 2;
[a, b] = [b, a];
console.log(a, b); // 2, 1

              

            


            

              
Object Destructuring

              

                
// Object destructuring
const user = {
    name: 'John',
    age: 30,
    email: 'john@example.com',
    address: {
        city: 'New York',
        country: 'USA'
    }
};

// Basic destructuring
const { name, age } = user;
console.log(name, age); // 'John', 30

// Alias variables
const { name: userName, age: userAge } = user;
console.log(userName, userAge); // 'John', 30

// Default values
const { name, role = 'user' } = user;
console.log(name, role); // 'John', 'user'

// Nested destructuring
const { address: { city, country } } = user;
console.log(city, country); // 'New York', 'USA'

// Rest operator
const { name, ...userInfo } = user;
console.log(userInfo); // { age: 30, email: '...', address: {...} }

// Function parameters
function greet({ name, age }) {
    return `Hello ${name}, you are ${age} years old`;
}

console.log(greet(user)); // 'Hello John, you are 30 years old'

              

            

          


          

            

               Spread & Rest Operators
            


            

              
Spread Operator Examples

              

                
// Array spreading
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];
const combined = [...arr1, ...arr2];
console.log(combined); // [1, 2, 3, 4, 5, 6]

// Copying arrays
const original = [1, 2, 3];
const copy = [...original];

// Adding elements
const withExtras = [...original, 4, 5];

// Object spreading
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };
const merged = { ...obj1, ...obj2 };
console.log(merged); // { a: 1, b: 2, c: 3, d: 4 }

// Function calls
function sum(a, b, c) {
    return a + b + c;
}
const numbers = [1, 2, 3];
console.log(sum(...numbers)); // 6

// String to array
const str = 'hello';
const chars = [...str];
console.log(chars); // ['h', 'e', 'l', 'l', 'o']

              

            


            

              
Rest Operator Examples

              

                
// Rest in function parameters
function sum(...numbers) {
    return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4)); // 10

// Rest with destructuring
const [first, second, ...rest] = [1, 2, 3, 4, 5];
console.log(first, second, rest); // 1, 2, [3, 4, 5]

const { a, b, ...others } = { a: 1, b: 2, c: 3, d: 4 };
console.log(a, b, others); // 1, 2, { c: 3, d: 4 }

// Rest in arrow functions
const multiply = (multiplier, ...numbers) =>
    numbers.map(num => num * multiplier);

console.log(multiply(2, 1, 2, 3)); // [2, 4, 6]

              

            

          


          

            

               Promises & Async/Await
            


            

              
Promise Creation and Usage

              

                
// Creating a Promise
const fetchData = () => {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            const success = Math.random() > 0.5;
            if (success) {
                resolve('Data fetched successfully');
            } else {
                reject('Failed to fetch data');
            }
        }, 1000);
    });
};

// Using Promises
fetchData()
    .then(result => {
        console.log('Success:', result);
        return result.toUpperCase();
    })
    .then(processed => {
        console.log('Processed:', processed);
    })
    .catch(error => {
        console.error('Error:', error);
    })
    .finally(() => {
        console.log('Operation completed');
    });

// Promise.all - run multiple promises in parallel
const promise1 = Promise.resolve('First');
const promise2 = new Promise(resolve => setTimeout(() => resolve('Second'), 1000));
const promise3 = fetch('https://api.example.com/data').then(res => res.json());

Promise.all([promise1, promise2, promise3])
    .then(results => console.log(results))
    .catch(error => console.error(error));

              

            


            

              
Async/Await Syntax

              

                
// Async function declaration
async function fetchUserData(userId) {
    try {
        const response = await fetch(`/api/users/${userId}`);
        if (!response.ok) {
            throw new Error('Failed to fetch user');
        }
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Error fetching user:', error);
        throw error; // Re-throw to allow caller to handle
    }
}

// Async arrow function
const fetchUserData = async (userId) => {
    try {
        const response = await fetch(`/api/users/${userId}`);
        const userData = await response.json();
        return userData;
    } catch (error) {
        console.error('Error:', error);
        return null; // Return default value on error
    }
};

// Using async functions
async function displayUser(userId) {
    try {
        const user = await fetchUserData(userId);
        if (user) {
            console.log('User:', user.name);
        }
    } catch (error) {
        console.error('Failed to display user:', error);
    }
}

// Top-level await (in modules)
const user = await fetchUserData(123);
console.log(user);

              

            

          


          

            

               Modules
            


            

              
Exporting from Modules

              

                
// math.js
export const PI = 3.14159;

export function add(a, b) {
    return a + b;
}

export function multiply(a, b) {
    return a * b;
}

// Default export
export default function square(x) {
    return x * x;
}

// Alternative default export syntax
export { square as default };

// Named exports
const subtract = (a, b) => a - b;
const divide = (a, b) => a / b;

export { subtract, divide };

              

            


            

              
Importing from Modules

              

                
// Importing named exports
import { add, multiply, PI } from './math.js';

// Importing default export
import square from './math.js';

// Importing with aliases
import { add as sum, multiply as product } from './math.js';

// Importing everything
import * as MathUtils from './math.js';
console.log(MathUtils.add(2, 3)); // 5

// Mixed imports
import square, { add, PI } from './math.js';

// Dynamic imports
const module = await import('./math.js');
console.log(module.add(1, 2));

 Modern Variable Declarations

          

            

              ES6 introduced let and const as
              alternatives to var, providing block scope and
              preventing common JavaScript pitfalls.
            


            

              

                
 var (ES5)

                

                  
// Function-scoped, can be redeclared
var name = 'John';
var name = 'Jane'; // No error
console.log(name); // 'Jane'

// Hoisting issues
console.log(age); // undefined (not error)
var age = 25;

// Loop scope problems
for (var i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100); // 3, 3, 3
}


                

                

                  
Issues with var:

                  
    
                    
  • Function-scoped (not block-scoped)
    • 
                    
  • Can be redeclared
    • 
                    
  • Hoisting behavior
    • 
                    
  • Closure loop problems
    • 
                  

                

              


              

                
 let & const (ES6)

                

                  
// Block-scoped
let name = 'John';
// let name = 'Jane'; // SyntaxError

const PI = 3.14159;
// PI = 3.14; // TypeError

// Proper scoping
for (let i = 0; i < 3; i++) {
    setTimeout(() => console.log(i), 100); // 0, 1, 2
}

// Temporal dead zone
// console.log(x); // ReferenceError
let x = 10;


                

                

                  
Benefits:

                  
    
                    
  • Block-scoped
    • 
                    
  • Cannot be redeclared
    • 
                    
  • No hoisting confusion
    • 
                    
  • Temporal dead zone protection
    •

 Arrow Functions

          

            

              Arrow functions provide a shorter syntax for writing functions and
              automatically bind the
              this context from the surrounding scope.
            


            

              

                
                
                
              


              

                
                

                  

                    

                      
Traditional Functions

                      

                        
// Regular function declaration
function add(a, b) {
    return a + b;
}

// Function expression
const multiply = function(a, b) {
    return a * b;
};

// Method in object
const calculator = {
    name: 'Calculator',
    add: function(a, b) {
        return a + b;
    }
};


                      

                    


                    

                      
Arrow Functions

                      

                        
// Basic arrow function
const add = (a, b) => {
    return a + b;
};

// Implicit return (single expression)
const multiply = (a, b) => a * b;

// Single parameter (parentheses optional)
const square = x => x * x;

// No parameters
const greet = () => 'Hello World!';

// Multiple statements
const processData = (data) => {
    const processed = data.map(item => item * 2);
    return processed.filter(item => item > 10);
};


                      

                    

                  

                


                
                

                  
Understanding 'this' in Arrow Functions

                  

                    
// Traditional function - 'this' depends on how it's called
const obj1 = {
    name: 'Object 1',
    regularFunction: function() {
        console.log('Regular function this:', this.name);
        
        setTimeout(function() {
            console.log('Timeout regular:', this.name); // undefined
        }, 1000);
    }
};

// Arrow function - 'this' inherited from enclosing scope
const obj2 = {
    name: 'Object 2',
    arrowFunction: function() {
        console.log('Regular function this:', this.name);
        
        setTimeout(() => {
            console.log('Timeout arrow:', this.name); // 'Object 2'
        }, 1000);
    }
};

// Event handlers comparison
class Button {
    constructor(element) {
        this.element = element;
        this.clickCount = 0;
        
        // Wrong way (this binding issue)
        this.element.addEventListener('click', function() {
            this.clickCount++; // 'this' refers to element, not Button
        });
        
        // Right way (arrow function preserves this)
        this.element.addEventListener('click', () => {
            this.clickCount++; // 'this' refers to Button instance
        });
    }
}


                  

                


                
                

                  
When to Use Arrow Functions


                  

                    

                      
 Good Use Cases

                      
    
                        
  • Array methods (map, filter, reduce)
    • 
                        
  • Event handlers in classes
    • 
                        
  • Callback functions
    • 
                        
  • Short, pure functions
    • 
                        
  • When you need lexical 'this'
    • 
                      

                      

                        
// Array methods
const numbers = [1, 2, 3, 4, 5];
const doubled = numbers.map(n => n * 2);
const evens = numbers.filter(n => n % 2 === 0);

// Callbacks
fetch('/api/data')
    .then(response => response.json())
    .then(data => console.log(data));


                      

                    


                    

                      
 Avoid When

                      
    
                        
  • Object methods (need dynamic 'this')
    • 
                        
  • Constructor functions
    • 
                        
  • Need 'arguments' object
    • 
                        
  • Methods that use 'call' or 'apply'
    • 
                        
  • Need function hoisting
    • 
                      

                      

                        
// Avoid for object methods
const user = {
    name: 'John',
    greet: () => {
        // 'this' doesn't refer to user object
        console.log(`Hello, ${this.name}`); // undefined
    }
};

// Use regular function instead
const user2 = {
    name: 'John',
    greet: function() {
        console.log(`Hello, ${this.name}`); // 'John'
    }
};

 Template Literals

          

            

              Template literals provide an easy way to create strings with
              embedded expressions, multi-line strings, and better formatting.
            


            

              

                

                  
Before ES6 (String Concatenation)

                  

                    
// String concatenation
var name = 'John';
var age = 30;
var greeting = 'Hello, my name is ' + name + 
               ' and I am ' + age + ' years old.';

// Multi-line strings (ugly)
var html = '<div class="card">' +
           '  <h2>' + title + '</h2>' +
           '  <p>' + description + '</p>' +
           '</div>';

// Expression evaluation
var total = 'Total: $' + (price * quantity).toFixed(2);


                  

                

              


              

                

                  
ES6+ (Template Literals)

                  

                    
// Template literals with expressions
const name = 'John';
const age = 30;
const greeting = `Hello, my name is ${name} and I am ${age} years old.`;

// Multi-line strings (clean)
const html = "
    

        
${title}

        
${description}

    

";

// Complex expressions
const total = `Total: $${(price * quantity).toFixed(2)}`;

// Tagged templates
const styled = css`
    color: ${primaryColor};
    font-size: ${fontSize}px;
`;

 Destructuring Assignment

          

            

              Destructuring allows you to extract values from arrays and objects
              into distinct variables in a more readable and concise way.
            


            

              

                
                
                
              


              

                
                

                  
Array Destructuring

                  

                    

                      
Basic Array Destructuring

                      

                        
// Traditional way
const colors = ['red', 'green', 'blue'];
const first = colors[0];
const second = colors[1];
const third = colors[2];

// ES6 destructuring
const [red, green, blue] = colors;
console.log(red);   // 'red'
console.log(green); // 'green'
console.log(blue);  // 'blue'

// Skip elements
const [primary, , tertiary] = colors;
console.log(primary);  // 'red'
console.log(tertiary); // 'blue'


                      

                    


                    

                      
Default Values & Rest

                      

                        
// Default values
const [a = 1, b = 2, c = 3] = [10];
console.log(a, b, c); // 10, 2, 3

// Rest operator
const numbers = [1, 2, 3, 4, 5];
const [first, second, ...rest] = numbers;
console.log(first);  // 1
console.log(second); // 2
console.log(rest);   // [3, 4, 5]

// Swapping variables
let x = 1, y = 2;
[x, y] = [y, x];
console.log(x, y); // 2, 1


                      

                    

                  

                


                
                

                  
Object Destructuring

                  

                    

                      
Basic Object Destructuring

                      

                        
// Traditional way
const user = {
    name: 'John Doe',
    age: 30,
    email: 'john@example.com'
};
const name = user.name;
const age = user.age;

// ES6 destructuring
const {name, age, email} = user;
console.log(name);  // 'John Doe'
console.log(age);   // 30
console.log(email); // 'john@example.com'

// Rename variables
const {name: userName, age: userAge} = user;
console.log(userName); // 'John Doe'
console.log(userAge);  // 30


                      

                    


                    

                      
Nested & Default Values

                      

                        
// Nested destructuring
const user = {
    name: 'John',
    address: {
        street: '123 Main St',
        city: 'Boston',
        country: 'USA'
    }
};

const {name, address: {city, country}} = user;
console.log(name);    // 'John'
console.log(city);    // 'Boston'
console.log(country); // 'USA'

// Default values
const {name, age = 25, city = 'Unknown'} = user;
console.log(age);  // 25 (default)
console.log(city); // 'Unknown' (default)


                      

                    

                  

                


                
                

                  
Advanced Destructuring Patterns

                  

                    

                      
Function Parameters

                      

                        
// Destructuring in function parameters
function greetUser({name, age, city = 'Unknown'}) {
    console.log(`Hello ${name}, you are ${age} years old from ${city}`);
}

greetUser({name: 'Alice', age: 28}); 
// "Hello Alice, you are 28 years old from Unknown"

// Array destructuring in parameters
function sum([a, b, c = 0]) {
    return a + b + c;
}

console.log(sum([1, 2]));    // 3
console.log(sum([1, 2, 3])); // 6


                      

                    


                    

                      
Mixed Destructuring

                      

                        
// Complex data structures
const response = {
    data: {
        users: [
            {id: 1, name: 'John', roles: ['admin', 'user']},
            {id: 2, name: 'Jane', roles: ['user']}
        ]
    },
    status: 'success'
};

// Extract nested data
const {
    data: {
        users: [
            {name: firstName, roles: [firstRole]},
            {name: secondName}
        ]
    },
    status
} = response;

console.log(firstName);  // 'John'
console.log(firstRole);  // 'admin'
console.log(secondName); // 'Jane'
console.log(status);     // 'success'

 Promises

          

            

              Promises represent the eventual completion (or failure) of an
              asynchronous operation and provide a cleaner alternative to
              callback-based code.
            


            

              

                

                  
                

                
Pending

                
Initial state, neither fulfilled nor rejected

              


              

                

                  
                

                
Fulfilled

                
Operation completed successfully

              


              

                

                  
                

                
Rejected

                
Operation failed

              

            


            

              
Promise Creation and Usage

              

                
                
                
              


              

                
                

                  

                    
// Creating a Promise
const myPromise = new Promise((resolve, reject) => {
    // Simulate async operation
    setTimeout(() => {
        const success = Math.random() > 0.5;
        
        if (success) {
            resolve('Operation successful!');
        } else {
            reject('Operation failed!');
        }
    }, 2000);
});

// Using the Promise
myPromise
    .then(result => {
        console.log('Success:', result);
    })
    .catch(error => {
        console.log('Error:', error);
    })
    .finally(() => {
        console.log('Promise completed');
    });

// Real-world example: API call
function fetchUserData(userId) {
    return new Promise((resolve, reject) => {
        // Simulate API call
        setTimeout(() => {
            if (userId > 0) {
                resolve({
                    id: userId,
                    name: 'John Doe',
                    email: 'john@example.com'
                });
            } else {
                reject(new Error('Invalid user ID'));
            }
        }, 1000);
    });
}

fetchUserData(1)
    .then(user => console.log('User:', user))
    .catch(error => console.log('Error:', error.message));


                  

                


                
                

                  

                    
// Promise chaining
fetchUserData(1)
    .then(user => {
        console.log('Got user:', user.name);
        // Return another promise
        return fetch(`/api/posts/${user.id}`);
    })
    .then(response => response.json())
    .then(posts => {
        console.log('User posts:', posts);
        // Chain another operation
        return posts.filter(post => post.published);
    })
    .then(publishedPosts => {
        console.log('Published posts:', publishedPosts);
    })
    .catch(error => {
        console.log('Error in chain:', error);
    });

// Avoiding callback hell
// Old way (callback hell)
getData(function(a) {
    getMoreData(a, function(b) {
        getEvenMoreData(b, function(c) {
            getFinalData(c, function(d) {
                // Finally done!
            });
        });
    });
});

// Promise way (much cleaner)
getData()
    .then(a => getMoreData(a))
    .then(b => getEvenMoreData(b))
    .then(c => getFinalData(c))
    .then(d => {
        // Finally done!
    })
    .catch(error => {
        // Handle any error in the chain
    });


                  

                


                
                

                  

                    
// Promise.all() - Wait for all promises
const promise1 = fetchUserData(1);
const promise2 = fetchUserData(2);
const promise3 = fetchUserData(3);

Promise.all([promise1, promise2, promise3])
    .then(users => {
        console.log('All users:', users);
        // All promises resolved
    })
    .catch(error => {
        console.log('At least one failed:', error);
    });

// Promise.allSettled() - Wait for all, regardless of outcome
Promise.allSettled([promise1, promise2, promise3])
    .then(results => {
        results.forEach((result, index) => {
            if (result.status === 'fulfilled') {
                console.log(`User ${index + 1}:`, result.value);
            } else {
                console.log(`User ${index + 1} failed:`, result.reason);
            }
        });
    });

// Promise.race() - First one to complete wins
Promise.race([promise1, promise2, promise3])
    .then(firstUser => {
        console.log('First user loaded:', firstUser);
    })
    .catch(error => {
        console.log('First promise failed:', error);
    });

// Promise.any() - First successful one wins
Promise.any([promise1, promise2, promise3])
    .then(firstSuccessfulUser => {
        console.log('First successful user:', firstSuccessfulUser);
    })
    .catch(error => {
        console.log('All promises failed:', error);
    });
