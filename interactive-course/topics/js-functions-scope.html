What are Functions?

Functions are one of the fundamental building blocks in JavaScript. A function is a reusable block of code designed to perform a particular task. Functions help you:

  • Organize code: Break complex problems into smaller, manageable pieces
  • Reuse code: Write once, use many times
  • Reduce repetition: Avoid duplicating code
  • Improve maintainability: Changes in one place affect all uses

Key Concept:

Think of functions like recipes - they take ingredients (parameters), follow steps (function body), and produce a result (return value).

Function Anatomy:

function functionName(parameter1, parameter2) {
  // Function body - code to execute
  let result = parameter1 + parameter2;
  return result; // Return value
}

Function Declaration vs Expression

Function Declaration

// Function Declaration
function greet(name) {
  return "Hello, " + name + "!";
}

// Can be called before declaration (hoisting)
console.log(greet("Alice")); // Works!

Features:

  • Hoisted (can call before declaration)
  • Creates a named function
  • Function statement

Function Expression

// Function Expression
const greet = function(name) {
  return "Hello, " + name + "!";
};

// Must be declared before use
console.log(greet("Bob")); // Works only after declaration

Features:

  • Not hoisted
  • Can be anonymous
  • Treated as a value

Arrow Function (ES6)

// Arrow Function
const greet = (name) => {
  return "Hello, " + name + "!";
};

// Short form for single expressions
const greetShort = name => "Hello, " + name + "!";

console.log(greetShort("Charlie"));

Features:

  • Shorter syntax
  • No 'this' binding
  • Implicit return for expressions

Parameters vs Arguments

Understanding the Difference

Parameters

Variables listed in the function definition

function add(a, b) { // a and b are parameters
  return a + b;
}

Arguments

Actual values passed when calling the function

add(5, 3); // 5 and 3 are arguments

Parameter Techniques

Default Parameters

function greet(name = "World") {
  return "Hello, " + name + "!";
}

console.log(greet());        // "Hello, World!"
console.log(greet("Alice")); // "Hello, Alice!"

Rest Parameters

function sum(...numbers) {
  let total = 0;
  for (let num of numbers) {
    total += num;
  }
  return total;
}

console.log(sum(1, 2, 3, 4, 5)); // 15

Destructuring Parameters

function createUser({name, age, email}) {
  return {
    name: name,
    age: age,
    email: email,
    id: Math.random()
  };
}

const user = createUser({
  name: "John",
  age: 30,
  email: "john@example.com"
});

Return Values

Understanding Return Statements

The return statement serves two purposes:

  • Ends function execution
  • Specifies the value to be returned to the caller

Returning Values

function multiply(a, b) {
  return a * b; // Returns the product
}

let result = multiply(4, 5);
console.log(result); // 20

Multiple Return Statements

function getGrade(score) {
  if (score >= 90) return "A";
  if (score >= 80) return "B";
  if (score >= 70) return "C";
  if (score >= 60) return "D";
  return "F";
}

No Return Statement

function sayHello(name) {
  console.log("Hello, " + name);
  // No return statement
}

let result = sayHello("Alice");
console.log(result); // undefined

Early Return

function divide(a, b) {
  if (b === 0) {
    return "Cannot divide by zero";
  }
  return a / b;
}

Variable Scope

Understanding Scope

Scope determines where variables can be accessed in your code. JavaScript has several types of scope:

Global Scope

Variables declared outside any function or block

let globalVar = "I'm global!";

function showGlobal() {
  console.log(globalVar); // Accessible here
}

console.log(globalVar); // Accessible here too

Function Scope

Variables declared inside a function

function myFunction() {
  let functionVar = "I'm local to this function";
  console.log(functionVar); // Accessible here
}

// console.log(functionVar); // Error! Not accessible here

Block Scope

Variables declared inside a block (with let/const)

if (true) {
  let blockVar = "I'm in a block";
  const alsoBlock = "Me too!";
  console.log(blockVar); // Accessible here
}

// console.log(blockVar); // Error! Not accessible here

let globalVar = "Global"; // Global scope

function outerFunction() {
  let functionVar = "Function"; // Function scope
  
  if (true) {
    let blockVar = "Block"; // Block scope
    
    // All variables accessible here
    console.log(globalVar);   // ✓
    console.log(functionVar); // ✓
    console.log(blockVar);    // ✓
  }
  
  // blockVar not accessible here
  console.log(globalVar);   // ✓
  console.log(functionVar); // ✓
  // console.log(blockVar); // ✗ Error!
}
