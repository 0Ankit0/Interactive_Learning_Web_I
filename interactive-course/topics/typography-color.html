Typography CSS Best Practices

/* CSS Custom Properties for Typography */
:root {
  /* Font Families */
  --font-primary: 'Inter', sans-serif;
  --font-heading: 'Playfair Display', serif;
  --font-mono: 'Fira Code', monospace;
  
  /* Font Sizes - Modular Scale */
  --text-xs: 0.75rem;    /* 12px */
  --text-sm: 0.875rem;   /* 14px */
  --text-base: 1rem;     /* 16px */
  --text-lg: 1.125rem;   /* 18px */
  --text-xl: 1.25rem;    /* 20px */
  --text-2xl: 1.5rem;    /* 24px */
  --text-3xl: 1.875rem;  /* 30px */
  --text-4xl: 2.25rem;   /* 36px */
  
  /* Line Heights */
  --leading-tight: 1.25;
  --leading-normal: 1.5;
  --leading-relaxed: 1.75;
  
  /* Letter Spacing */
  --tracking-tight: -0.025em;
  --tracking-normal: 0;
  --tracking-wide: 0.025em;
}

/* Typography Classes */
.text-heading {
  font-family: var(--font-heading);
  font-weight: 700;
  line-height: var(--leading-tight);
  letter-spacing: var(--tracking-tight);
}

.text-body {
  font-family: var(--font-primary);
  font-weight: 400;
  line-height: var(--leading-normal);
  letter-spacing: var(--tracking-normal);
}

/* Responsive Typography */
h1 {
  font-size: clamp(var(--text-2xl), 4vw, var(--text-4xl));
}

/* Improve Readability */
p {
  max-width: 65ch; /* Optimal line length */
  margin-bottom: 1.5em;
}

/* Web Font Loading */
@font-face {
  font-family: 'Inter';
  src: url('inter.woff2') format('woff2');
  font-display: swap; /* Improve loading performance */
}

/* CSS Custom Properties for Colors */
:root {
  /* Color Palette */
  --color-primary: hsl(210, 79%, 46%);
  --color-primary-dark: hsl(210, 79%, 36%);
  --color-primary-light: hsl(210, 79%, 56%);
  
  --color-secondary: hsl(142, 69%, 49%);
  --color-accent: hsl(6, 78%, 57%);
  
  /* Neutral Colors */
  --color-white: hsl(0, 0%, 100%);
  --color-gray-50: hsl(210, 20%, 98%);
  --color-gray-100: hsl(210, 16%, 93%);
  --color-gray-300: hsl(210, 14%, 83%);
  --color-gray-500: hsl(210, 11%, 71%);
  --color-gray-700: hsl(210, 10%, 40%);
  --color-gray-900: hsl(210, 24%, 16%);
  
  /* Semantic Colors */
  --color-success: hsl(142, 69%, 49%);
  --color-warning: hsl(38, 92%, 50%);
  --color-error: hsl(6, 78%, 57%);
  --color-info: hsl(199, 89%, 48%);
  
  /* Text Colors */
  --color-text-primary: var(--color-gray-900);
  --color-text-secondary: var(--color-gray-700);
  --color-text-muted: var(--color-gray-500);
}

/* Dark Mode Support */
@media (prefers-color-scheme: dark) {
  :root {
    --color-text-primary: var(--color-gray-50);
    --color-text-secondary: var(--color-gray-300);
    --color-text-muted: var(--color-gray-500);
  }
}

/* Color Utility Classes */
.text-primary { color: var(--color-primary); }
.bg-primary { background-color: var(--color-primary); }
.border-primary { border-color: var(--color-primary); }

/* Accessibility Considerations */

  height: 1px;
  padding: 0;
  margin: -1px;
  overflow: hidden;
  clip: rect(0, 0, 0, 0);
  white-space: nowrap;
  border: 0;
}

.responsive-text {

  line-height: clamp(1.2, 1.5, 1.8);
}

/* Container Queries for Typography */
@container (min-width: 400px) {
  .card-title {
    font-size: var(--text-xl);
  }
}

/* High Contrast Mode Support */
@media (prefers-contrast: high) {
  :root {
    --color-primary: hsl(210, 100%, 30%);
    --color-text-primary: hsl(0, 0%, 0%);
  }
}

/* Reduced Motion Support */
@media (prefers-reduced-motion: reduce) {
  * {
    animation-duration: 0.01ms !important;
    animation-iteration-count: 1 !important;
    transition-duration: 0.01ms !important;
  }
}

/* Print Styles */
@media print {
  :root {
    --color-primary: hsl(0, 0%, 0%);
    --color-text-primary: hsl(0, 0%, 0%);
  }
  
  .bg-primary {
    background: transparent !important;
    border: 1px solid black;
  }
}
