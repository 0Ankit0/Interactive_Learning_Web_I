Introduction to Error Handling

Why Error Handling Matters

Error handling is crucial for building robust applications that can gracefully handle unexpected situations. Without proper error handling, applications can crash, provide poor user experience, or expose sensitive information.

Benefits of Error Handling

  • Prevents application crashes
  • Provides better user experience
  • Helps with debugging and maintenance
  • Improves application reliability
  • Allows graceful degradation

Common Error Scenarios

  • Network failures
  • Invalid user input
  • File system errors
  • API failures
  • Memory issues

Try-Catch Blocks

Basic Try-Catch Syntax

try {
    // Code that might throw an error
    const result = riskyOperation();
    console.log('Success:', result);
} catch (error) {
    // Handle the error
    console.error('Error occurred:', error.message);
} finally {
    // Always executed
    console.log('Cleanup code here');
}

               Throwing Custom Errors
            


            

              
Creating Custom Errors

              

                
// Throwing custom errors
function validateAge(age) {
    if (age < 0) {
        throw new Error('Age cannot be negative');
    }
    if (age > 150) {
        throw new RangeError('Age seems unrealistic');
    }
    return true;
}

// Using custom error types
class ValidationError extends Error {
    constructor(message) {
        super(message);
        this.name = 'ValidationError';
    }
}

function validateEmail(email) {
    if (!email.includes('@')) {
        throw new ValidationError('Invalid email format');
    }
}

              

            

          


          

            

               Async Error Handling
            


            

              
Promises with Error Handling

              

                
// Promise error handling
fetch('/api/data')
    .then(response => {
        if (!response.ok) {
            throw new Error('Network response was not ok');
        }
        return response.json();
    })
    .then(data => console.log(data))
    .catch(error => console.error('Fetch error:', error));

// Async/await error handling
async function fetchData() {
    try {
        const response = await fetch('/api/data');
        if (!response.ok) {
            throw new Error('Network response was not ok');
        }
        const data = await response.json();
        return data;
    } catch (error) {
        console.error('Async error:', error);
        throw error; // Re-throw if needed
    }
}

                 Understanding JavaScript
                Errors
              
    
              
    
                
    
                  Error handling is a crucial aspect of JavaScript programming
                  that allows you to anticipate, catch, and manage runtime
                  errors gracefully. Proper error handling prevents applications
                  from crashing and provides users with meaningful feedback.
                
    

                
    
                  
    
                    
    
                      
                    
    
                    
    Syntax Errors
    
                    
    
                      Errors in code structure that prevent parsing and
                      execution.
                    
    
                    
    
                      
    // Syntax Error - Missing closing parenthesis
console.log("Hello World";
    
                    
    
                  
    

                  
    
                    
    
                      
                    
    
                    
    Runtime Errors
    
                    
    
                      Errors that occur during code execution, such as accessing
                      undefined properties.
                    
    
                    
    
                      
    // Runtime Error - Cannot read property of undefined
let obj;
console.log(obj.name); // TypeError
    
                    
    
                  
    

                  
    
                    
    
                      
                    
    
                    
    Logical Errors
    
                    
    
                      Code runs without throwing errors but produces incorrect
                      results.
                    
    
                    
    
                      
    // Logical Error - Wrong operator
function add(a, b) {
  return a - b; // Should be a + b
}
    
                    
    
                  
    

                  
    
                    
    
                      
                    
    
                    
    Network Errors
    
                    
    
                      Errors related to network requests, API calls, or
                      connectivity issues.
                    
    
                    
    
                      
    // Network Error - Failed fetch request
fetch('/api/data')
  .catch(error => console.error('Network error:', error));

                 Async Error Handling
              
    
              
    
                
    
                  Handling errors in asynchronous code requires special
                  attention, especially with Promises and async/await.
                
    

                
    
                  
    
                    
    Promise Error Handling
    
                    
    
                      
    // Using .catch() with Promises
fetch('/api/data')
  .then(response => response.json())
  .then(data => console.log(data))
  .catch(error => {
    console.error('Error fetching data:', error);
    // Handle error (show user message, retry, etc.)
  });

// Chaining multiple catches for different error types
fetch('/api/data')
  .then(response => {
    if (!response.ok) {
      throw new Error(`HTTP ${response.status}: ${response.statusText}`);
    }
    return response.json();
  })
  .catch(error => {
    if (error instanceof TypeError) {
      console.error('Network error:', error);
    } else {
      console.error('HTTP error:', error);
    }
  });
    
                    
    
                  
    

                  
    
                    
    Async/Await Error Handling
    
                    
    
                      
    async function fetchUserData(userId) {
  try {
    const response = await fetch(`/api/users/${userId}`);
    
    if (!response.ok) {
      throw new Error(`Failed to fetch user: ${response.status}`);
    }
    
    const userData = await response.json();
    return userData;
    
  } catch (error) {
    if (error.name === 'TypeError') {
      // Network error
      console.error('Network connection failed:', error);
      throw new Error('Unable to connect to server');
    } else {
      // Other errors
      console.error('Error fetching user data:', error);
      throw error;
    }
  }
}

// Usage with error handling
try {
  const user = await fetchUserData(123);
  console.log('User data:', user);
} catch (error) {
  displayErrorMessage(error.message);
}

    Use Error Boundaries
    
                  
    
                    Implement centralized error handling to catch unhandled

    window.addEventListener('error', (event) => {
  console.error('Global error:', event.error);
  // Send to logging service
  logError(event.error);
});

window.addEventListener('unhandledrejection', (event) => {
  console.error('Unhandled promise rejection:', event.reason);
  event.preventDefault();
});
    
                  
    
                
    

                
    
                  
    
                  
    Implement Retry Logic
    
                  
    
                    For network errors, implement retry mechanisms with
                    exponential backoff.
                  
    
                  
    
                    
    async function fetchWithRetry(url, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fetch(url);
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => 
        setTimeout(resolve, 1000 * Math.pow(2, i))
      );
    }
  }
}
    
                  
    
                
    

                
    
                  
    
                    
                  
    
                  
    User-Friendly Messages
    
                  
    
                    Show users helpful error messages, not technical details.
                  
    
                  
    
                    
    function handleError(error) {
  // Log technical details for developers
  console.error('Technical error:', error);
  
  // Show user-friendly message
  const userMessage = getUserFriendlyMessage(error);
  showNotification(userMessage, 'error');
}

function getUserFriendlyMessage(error) {
  if (error.name === 'NetworkError') {
    return 'Unable to connect. Please check your internet connection.';
  }
  return 'Something went wrong. Please try again later.';
}

  .catch(error => {
    if (error instanceof TypeError) {
      console.error('Network error:', error);
    } else {
      console.error('HTTP error:', error);
    }
  });
    
                
    
              
    

              
    
                
    Async/Await Error Handling
    
                
    
                  
    async function fetchUserData(userId) {
  try {
    const response = await fetch(`/api/users/${userId}`);
    
    if (!response.ok) {
      throw new Error(`Failed to fetch user: ${response.status}`);
    }
    
    const userData = await response.json();
    return userData;
    
  } catch (error) {
    if (error.name === 'TypeError') {
      // Network error
      console.error('Network connection failed:', error);
      throw new Error('Unable to connect to server');
    } else {
      // Other errors
      console.error('Error fetching user data:', error);
      throw error;
    }
  }
}

// Usage with error handling
try {
  const user = await fetchUserData(123);
  console.log('User data:', user);
} catch (error) {
  displayErrorMessage(error.message);
}

    Implement Retry Logic
    
              
    
                For network errors, implement retry mechanisms with exponential
                backoff.
              
    
              
    
                
    async function fetchWithRetry(url, maxRetries = 3) {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fetch(url);
    } catch (error) {
      if (i === maxRetries - 1) throw error;
      await new Promise(resolve => 
        setTimeout(resolve, 1000 * Math.pow(2, i))
      );
    }
  }
}
    
              
    
            
    

            
    
              
    
                
              
    
              
    User-Friendly Messages
    
              
    Show users helpful error messages, not technical details.
    
              
    
                
    function handleError(error) {
  // Log technical details for developers
  console.error('Technical error:', error);
  
  // Show user-friendly message
  const userMessage = getUserFriendlyMessage(error);
  showNotification(userMessage, 'error');
}

function getUserFriendlyMessage(error) {
  if (error.name === 'NetworkError') {
    return 'Unable to connect. Please check your internet connection.';
  }
  return 'Something went wrong. Please try again later.';
}
