What is jQuery?

jQuery is a fast, lightweight, and feature-rich JavaScript library. It makes things like HTML document traversal and manipulation, event handling, animation, and Ajax much simpler with an easy-to-use API that works across different browsers.

Vanilla JavaScript

const elements = document.querySelectorAll('.btn');
for (let i = 0; i < elements.length; i++) {
    elements[i].addEventListener('click', function() {
        this.style.backgroundColor = 'blue';
    });
}
8 lines of code
VS

jQuery

$(document).ready(function() {
    $('.btn').click(function() {
        $(this).css('background-color', 'blue');
    });
});
4 lines of code

jQuery Selectors

jQuery selectors allow you to select and manipulate HTML elements. They are based on CSS selectors and provide additional custom selectors.

Basic Selectors

// Element selector
$('p')  // Selects all <p> elements

// ID selector
$('#myId')  // Selects element with id="myId"

// Class selector
$('.myClass')  // Selects all elements with class="myClass"

// Universal selector
$('*')  // Selects all elements

Advanced Selectors

// Attribute selector
$('[href]')  // Elements with href attribute
$('[href="#"]')  // href equals "#"

// Pseudo selectors
$('li:first')  // First li element
$('li:last')   // Last li element
$('li:even')   // Even li elements
$('tr:odd')    // Odd tr elements

// Form selectors
$(':input')    // All input elements
$(':checked')  // Checked checkboxes/radios

DOM Manipulation

jQuery makes it easy to manipulate the DOM - get and set content, add/remove elements, and modify attributes.

Content Methods

// Get/Set text content
$('#myDiv').text()           // Get text
$('#myDiv').text('Hello')    // Set text

// Get/Set HTML content
$('#myDiv').html()           // Get HTML
$('#myDiv').html('<b>Bold</b>')  // Set HTML

// Get/Set value
$('#myInput').val()          // Get value
$('#myInput').val('text')    // Set value

Add/Remove Elements

// Append content
$('#myDiv').append('<p>New</p>')
$('#myDiv').prepend('<p>First</p>')

// Insert content
$('#myDiv').before('<p>Before</p>')
$('#myDiv').after('<p>After</p>')

// Remove elements
$('#myDiv').remove()         // Remove element
$('#myDiv').empty()          // Remove children

Attributes & Properties

// Attributes
$('#myLink').attr('href')           // Get
$('#myLink').attr('href', 'url')    // Set
$('#myLink').removeAttr('href')     // Remove

// CSS Classes
$('#myDiv').addClass('active')
$('#myDiv').removeClass('active')
$('#myDiv').toggleClass('active')
$('#myDiv').hasClass('active')      // Check

CSS Styling

// Get/Set single CSS property
$('#myDiv').css('color')              // Get
$('#myDiv').css('color', 'red')       // Set

// Set multiple properties
$('#myDiv').css({
    'color': 'red',
    'background-color': 'blue',
    'font-size': '20px'
});

Event Handling

jQuery provides simple methods to attach event handlers to elements and respond to user interactions.

Mouse Events

// Click event
$('#myBtn').click(function() {
    alert('Button clicked!');
});

// Double click
$('#myBtn').dblclick(function() {
    console.log('Double clicked');
});

// Hover (mouseenter + mouseleave)
$('#myDiv').hover(
    function() { $(this).addClass('hover'); },
    function() { $(this).removeClass('hover'); }
);

Keyboard Events

// Keypress event
$('#myInput').keypress(function(e) {
    console.log('Key pressed: ' + e.key);
});

// Keydown event
$('#myInput').keydown(function(e) {
    if (e.key === 'Enter') {
        console.log('Enter key pressed');
    }
});

// Keyup event
$('#myInput').keyup(function() {
    var value = $(this).val();
    console.log('Current value: ' + value);
});

Form Events

// Submit event
$('#myForm').submit(function(e) {
    e.preventDefault();  // Prevent default submission
    console.log('Form submitted');
});

// Focus and blur events
$('#myInput').focus(function() {
    $(this).css('background-color', '#ffffcc');
});

$('#myInput').blur(function() {
    $(this).css('background-color', '#ffffff');
});

Event Delegation

// Attach event to parent for dynamic elements
$('#parent').on('click', '.child', function() {
    console.log('Child clicked');
});

// Multiple events
$('#myDiv').on({
    mouseenter: function() { $(this).addClass('hover'); },
    mouseleave: function() { $(this).removeClass('hover'); },
    click: function() { $(this).toggleClass('active'); }
});

AJAX with jQuery

jQuery simplifies AJAX requests, making it easy to load data from the server without page refresh.

GET Request

// Simple GET request
$.get('/api/users', function(data) {
    console.log(data);
});

// GET with parameters
$.get('/api/users', { page: 1, limit: 10 }, function(data) {
    console.log(data);
});

// Using $.ajax()
$.ajax({
    url: '/api/users',
    method: 'GET',
    success: function(data) {
        console.log(data);
    },
    error: function(error) {
        console.error(error);
    }
});

POST Request

// Simple POST request
$.post('/api/users', { name: 'John', email: 'john@example.com' }, function(data) {
    console.log('User created:', data);
});

// Using $.ajax()
$.ajax({
    url: '/api/users',
    method: 'POST',
    data: JSON.stringify({ name: 'John' }),
    contentType: 'application/json',
    success: function(data) {
        console.log(data);
    }
});

Load Content

// Load HTML into element
$('#myDiv').load('/content.html');

// Load with selector
$('#myDiv').load('/page.html #content');

// Load with callback
$('#myDiv').load('/content.html', function(response, status) {
    if (status === 'success') {
        console.log('Content loaded');
    }
});

JSON Request

// Get JSON data
$.getJSON('/api/data.json', function(data) {
    $.each(data, function(index, item) {
        console.log(item.name);
    });
});

// Get JSON with parameters
$.getJSON('/api/posts', { userId: 1 }, function(posts) {
    console.log(posts);
});

Animations & Effects

jQuery provides built-in animation methods to create smooth visual effects and transitions.

Show/Hide Effects

// Show/Hide instantly
$('#myDiv').show();
$('#myDiv').hide();

// Show/Hide with duration
$('#myDiv').show(1000);    // 1 second
$('#myDiv').hide('slow');  // slow, fast, or milliseconds

// Toggle visibility
$('#myDiv').toggle();
$('#myDiv').toggle(500);

Slide Effects

// Slide down (show with sliding)
$('#myDiv').slideDown();
$('#myDiv').slideDown(1000);

// Slide up (hide with sliding)
$('#myDiv').slideUp();
$('#myDiv').slideUp('fast');

// Toggle sliding
$('#myDiv').slideToggle();
$('#myDiv').slideToggle(500);

Fade Effects

// Fade in
$('#myDiv').fadeIn();
$('#myDiv').fadeIn(1000);

// Fade out
$('#myDiv').fadeOut();
$('#myDiv').fadeOut('slow');

// Fade to specific opacity
$('#myDiv').fadeTo(1000, 0.5);  // 50% opacity

// Toggle fade
$('#myDiv').fadeToggle();

Custom Animations

// Animate CSS properties
$('#myDiv').animate({
    left: '250px',
    opacity: 0.5,
    height: '150px',
    width: '150px'
}, 1000);

// With easing and callback
$('#myDiv').animate({
    left: '250px'
}, {
    duration: 1000,
    easing: 'swing',
    complete: function() {
        console.log('Animation complete');
    }
});

Practical Examples

Real-world examples showing how to use jQuery in common web development scenarios.

Accordion Menu

$(document).ready(function() {
    $('.accordion-header').click(function() {
        // Close all other panels
        $('.accordion-content').not($(this).next()).slideUp();
        
        // Toggle clicked panel
        $(this).next('.accordion-content').slideToggle();
        
        // Toggle active class
        $(this).toggleClass('active');
        $('.accordion-header').not(this).removeClass('active');
    });
});

Image Gallery

$(document).ready(function() {
    $('.gallery-thumb').click(function() {
        var imgSrc = $(this).attr('src');
        var fullSrc = imgSrc.replace('thumb', 'full');
        
        $('#lightbox-img').attr('src', fullSrc);
        $('#lightbox').fadeIn();
    });
    
    $('.close-lightbox').click(function() {
        $('#lightbox').fadeOut();
    });
});

Form Validation

$(document).ready(function() {
    $('#myForm').submit(function(e) {
        var isValid = true;
        
        // Validate email
        var email = $('#email').val();
        if (!email.includes('@')) {
            $('#email').addClass('error');
            isValid = false;
        }
        
        // Validate required fields
        $('[required]').each(function() {
            if ($(this).val() === '') {
                $(this).addClass('error');
                isValid = false;
            }
        });
        
        if (!isValid) {
            e.preventDefault();
        }
    });
});

Mobile Menu Toggle

$(document).ready(function() {
    $('.menu-toggle').click(function() {
        $('.mobile-menu').slideToggle();
        $(this).toggleClass('active');
        
        // Animate hamburger icon
        $(this).find('.bar').toggleClass('change');
    });
    
    // Close menu on outside click
    $(document).click(function(e) {
        if (!$(e.target).closest('.menu-toggle, .mobile-menu').length) {
            $('.mobile-menu').slideUp();
            $('.menu-toggle').removeClass('active');
        }
    });
});
